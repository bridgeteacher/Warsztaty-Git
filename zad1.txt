using System;

class Program
{
    static void Main()
    {
        Console.Write("Enter the limit for Fibonacci series: ");
        int limit = int.Parse(Console.ReadLine());
        
        Console.Write("Fib2 Series master with rebase:");
        PrintFibonacci(limit);
    }

    static void PrintFibonacci(int limit)
    {
        int a = 0, b = 1, c;
        Console.Write(a + " " + b + " ");
        
        while (true)
        {
            c = a + b;
            if (c > limit) break;
            Console.Write(c + " ");
            a = b;
            b = c;
        }
    }
}
